###                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                方法

-                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                `loadGeoJSON(geoData:                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                any)`:                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                加载并渲染                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                GeoJSON                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                数据
-                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                `updateStyle(style:                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                Partial<MapStyle>)`:                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                更新地图样式
-                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                `reset()`:                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                重置视图到初始状态
-                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                `render()`:                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                手动触发重新渲染
-                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                `destroy()`:                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                销毁渲染器实例
