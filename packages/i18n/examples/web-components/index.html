Basic Translation Components

Text Component

<i18n-text key="common.ok"></i18n-text>
<i18n-text key="missing.key" default="Default Text"></i18n-text>

Button Component

<i18n-button text-key="common.save" variant="primary"></i18n-button>
<i18n-button text-key="common.delete" variant="danger" size="small"></i18n-button>

Input Component

<i18n-input
  label-key="validation.username.required"
  placeholder-key="common.searchPlaceholder"
  type="text">
</i18n-input>

Parameter Interpolation

Dynamic Parameters

<i18n-text key="common.pageOf" params='{"current": 1, "total": 10}'></i18n-text>
<i18n-text key="common.showingItems" params='{"start": 1, "end": 20, "total": 100}'></i18n-text>

Button with Parameters

<i18n-button
  text-key="common.pageOf"
  params='{"current": 2, "total": 5}'
  variant="secondary">
</i18n-button>

Event Handling

Component Events

Event Log:
Click button or type in input to see events...
// Listen to component events
document.querySelector('i18n-button').addEventListener('click', (e) => {
  console.log('Button clicked:', e.detail)
})

document.querySelector('i18n-language-switcher').addEventListener('language-changed', (e) => {
  console.log('Language changed:', e.detail)
})
