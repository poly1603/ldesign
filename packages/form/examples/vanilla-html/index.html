快速开始

1. 创建表单实例

import { createForm } from '@ldesign/form';

const form = createForm({
  initialValues: {
    name: '',
    email: ''
  }
});

2. 注册字段

const nameField = form.registerField({
  name: 'name',
  rules: [
    { required: true, message: '姓名是必填项' }
  ]
});

3. 处理表单提交

form.onSubmit((event) => {
  if (event.valid) {
    console.log('表单数据:', event.data);
  } else {
    console.log('验证失败:', event.errors);
  }
});
