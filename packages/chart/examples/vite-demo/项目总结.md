### 集成方式
```javascript
// React 项目中使用
import { Chart } from '@ldesign/chart'

function MyComponent() {
  useEffect(() => {
    const chart = new Chart(containerRef.current, config)
    return () => chart.dispose()
  }, [])
  
  return <div ref={containerRef} />
}
```
